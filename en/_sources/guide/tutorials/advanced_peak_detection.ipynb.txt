    "# 擬似データの作成\n",
    "df = 0.5 * u.Hz\n",
    "f_axis = np.arange(0, 500, df.value) * u.Hz\n",
    "data = np.exp(-f_axis.value / 100) + np.random.normal(0, 0.05, len(f_axis))\n",
    "\n",
    "# ピークを追加\n",
    "peak_freqs_true = [50, 60, 120, 125, 300] * u.Hz\n",
    "for f in peak_freqs_true:\n",
    "    idx = int(f.value / df.value)\n",
    "    data[idx] += 1.0\n",
    "\n",
    "spec = FrequencySeries(data, df=df, unit=\"V/Hz\")\n",
    "\n",
    "# --- 基本的な検出 (しきい値のみ) ---\n",
    "# peaks は FrequencySeries オブジェクトとして返ってきます\n",
    "peaks, _ = spec.find_peaks(threshold=0.5)\n",
    "\n",
    "plt.figure(figsize=(10, 5))\n",
    "plt.plot(spec, label=\"Spectrum\", alpha=0.7)\n",
    "plt.plot(peaks, \"x\", color=\"red\", label=\"Basic Peaks\")\n",
    "plt.title(\"Basic Peak Detection\")\n",
    "plt.xlabel(\"Frequency [Hz]\"); plt.ylabel(\"Amplitude [V/Hz]\")\n",
    "plt.legend(); plt.show()\n"
